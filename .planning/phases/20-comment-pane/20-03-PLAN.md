```typescript
export function offsetFromPoint(
  container: HTMLElement,
  clientX: number,
  clientY: number,
): number | null
// Implementation:
//   1. Prefer document.caretRangeFromPoint(clientX, clientY) (WebKit/Blink)
//   2. Fall back to document.caretPositionFromPoint(clientX, clientY) (Firefox)
//   3. If the resulting node is not a Text node OR not a descendant of `container`, return null
//   4. Compute the character offset relative to `container` by walking previous text nodes within container
//      and summing their .length, then adding the node-local offset. Mirror getRangeOffsets() in useTextSelection.ts.
//   5. Return that integer (or null on any failure).
```

<!-- Reverse-direction hover logic in ContentPane (COMMENT-02 second direction) -->
```typescript
function handleMouseMove(e: React.MouseEvent<HTMLDivElement>) {
  if (!onHoverCommentId || !annotations || !planRef.current) return
  const offset = offsetFromPoint(planRef.current, e.clientX, e.clientY)
  if (offset === null) {
    onHoverCommentId(null)
    return
  }
  const hit = annotations.find(
    (a) => offset >= a.anchorStart && offset < a.anchorEnd,
  )
  onHoverCommentId(hit ? hit.id : null)
}
```
