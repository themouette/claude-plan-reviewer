    5. `buildCodeReviewPayload('changes_requested', [lineFixture])` where `lineFixture = { id: 'x', type: 'line', file: 'src/foo.ts', side: 'additions', lineNumber: 42, text: 'Use const here', createdAt: '2026-05-25T00:00:00Z' }` -> parsed.decision === `'changes_requested'`; parsed.comments[0] equals `{ file: 'src/foo.ts', line: 42, side: 'additions', text: 'Use const here' }`.
