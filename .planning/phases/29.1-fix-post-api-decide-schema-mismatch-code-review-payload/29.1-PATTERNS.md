```rust
async fn post_decide(
    State(state): State<Arc<AppState>>,
    Json(body): Json<serde_json::Value>,   // accept any JSON object
) -> impl IntoResponse {
    // Discriminate by key presence:
    // Plan-review frontend sends: { behavior: "allow"|"deny", message?: "..." }
    // Code-review frontend sends: { message?: "...", comments?: [...] }
    let decision = if let Some(behavior) = body.get("behavior").and_then(|v| v.as_str()) {
        // Plan-review shape — use behavior as provided
        Decision {
            behavior: behavior.to_string(),
            message: body.get("message").and_then(|v| v.as_str()).map(str::to_string),
        }
    } else {
        // Code-review shape — behavior is implicitly "allow"
        Decision {
            behavior: "allow".to_string(),
            message: body.get("message").and_then(|v| v.as_str()).map(str::to_string),
        }
    };
    let tx = state.decision_tx.lock().unwrap().take();
    match tx {
        Some(tx) => {
            let _ = tx.send(decision);
            StatusCode::OK
        }
        None => StatusCode::CONFLICT,
    }
}
```

**Key change:** `Json<Decision>` becomes `Json<serde_json::Value>`. The `Decision` type and `AppState` are unchanged. The router registration is unchanged.

**New spawn helper to add (copy `spawn_hook_flow` structure, change args + stdin):**
```rust
/// Spawn the binary in code-review mode (no stdin — the subcommand reads no hook input).
fn spawn_code_review_flow(port: u16) -> std::process::Child {
    let home = Box::leak(Box::new(tempfile::TempDir::new().unwrap()));
    Command::new(binary_path())
        .env("HOME", home.path())
        .args(["--no-browser", "--port", &port.to_string(), "code-review"])
        .stdin(Stdio::null())    // code-review reads no stdin
        .stdout(Stdio::piped())
        .stderr(Stdio::piped())
        .spawn()
        .expect("failed to spawn plan-reviewer code-review")
}
```
