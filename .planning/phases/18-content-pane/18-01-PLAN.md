import { marked } from 'marked'

import 'highlight.js/styles/github-dark.css'

marked.use(markedHighlight({
  langPrefix: 'hljs language-',
  highlight(code, lang) {
    const language = hljs.getLanguage(lang) ? lang : 'plaintext'
    return hljs.highlight(code, { language }).value
  },
}))
marked.use({ gfm: true })

// Usage at call site: marked.parse(planMd) as string

| Server → Browser (`GET /api/plan`) | Untrusted markdown content from the plan file crosses into the browser; consumed by `renderMarkdown` and then injected via `dangerouslySetInnerHTML` in Plan 03. This plan does not perform the injection but produces the parsed HTML used later. |
| Browser DOM → JS (text selection) | User-controlled selection text reaches `useTextSelection` via `document.getSelection()`. Only character offsets and the trimmed text are persisted. |
